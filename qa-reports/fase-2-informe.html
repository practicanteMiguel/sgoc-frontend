1
API calls directas con axios en componente (2 mutaciones)
src/components/modulos/consumables/compras/informe-compras-tab.tsx — lineas 273, 353
CRITICO Tipo: Separacion de capas Impacto: Testing, reusabilidad

Las funciones saveOC() (linea 265) y saveAll() (linea 340) llaman directamente a api.patch() dentro del componente, ademas de usar useQueryClient para invalidar el cache manualmente. Esto duplica la logica de mutacion que deberia vivir en el hook use-compras.ts. 

// Antes - api.patch directo en el componente (informe-compras-tab.tsx:265)
const qc = useQueryClient()

async function saveOC() {
  await Promise.all(entries.map(([rq_id, item]) =>
    api.patch(`/requisiciones/${rq_id}/facturas`, { items: [...] })
  ))
  qc.invalidateQueries({ queryKey: ['requisiciones'] })
  qc.invalidateQueries({ queryKey: ['compras-informe'] })
}

// Despues - hook dedicado en use-compras.ts
export function useSaveFacturas() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (entries: FacturaEntry[]) =>
      Promise.all(entries.map(([rq_id, item]) =>
        api.patch(`/requisiciones/${rq_id}/facturas`, { items: [...] })
      )),
    onSuccess: () => {
      qc.invalidateQueries({ queryKey: ['requisiciones'] })
      qc.invalidateQueries({ queryKey: ['compras-informe'] })
    },
  })
}
2
api.delete directo + useQueryClient en componente de notificaciones
src/components/modulos/settings/notifications-tab.tsx — lineas 145, 152
CRITICO Tipo: Separacion de capas

handleDelete() llama a api.delete() directamente y luego invalida el cache con qc.invalidateQueries(). El hook use-notifications.ts existe y ya maneja notificaciones pero no tiene una mutacion de eliminacion individual. 

// Antes (notifications-tab.tsx:150)
const qc = useQueryClient()
const handleDelete = async (id: string) => {
  try {
    await api.delete(`/notifications/${id}`)
    qc.invalidateQueries({ queryKey: ['notifications'] })
  } catch { /* silencioso */ }
}

// Despues - agregar useMutation en use-notifications.ts
export function useDeleteNotification() {
  const qc = useQueryClient()
  return useMutation({
    mutationFn: (id: string) => api.delete(`/notifications/${id}`),
    onSuccess: () => qc.invalidateQueries({ queryKey: ['notifications'] }),
    onError: () => toast.error('No se pudo eliminar la notificacion'),
  })
}

// Agregar en src/lib/utils.ts
export function formatCOP(value: number | string | null | undefined): string {
  if (value === null || value === undefined || value === '') return '-'
  const num = typeof value === 'string' ? parseFloat(value) : value
  if (isNaN(num)) return '-'
  return new Intl.NumberFormat('es-CO', {
    style: 'currency', currency: 'COP', maximumFractionDigits: 0,
  }).format(num)
}

// Agregar en src/lib/utils.ts
export function formatDateShort(iso: string): string {
  const d = new Date(iso.includes('T') ? iso : iso + 'T00:00:00')
  if (isNaN(d.getTime())) return '-'
  return d.toLocaleDateString('es-CO', { day: '2-digit', month: 'short', year: 'numeric' })
}
